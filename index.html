        function Flashcards() {

          const [index, setIndex] = useState(0);

          const [isFlipped, setIsFlipped] = useState(false);


          const word = vocabList[index];


          const handleNext = () => {

            setIsFlipped(false);

            setTimeout(() => setIndex((prev) => (prev + 1) % vocabList.length), 150);

          };


          const handlePrev = () => {

            setIsFlipped(false);

            setTimeout(() => setIndex((prev) => (prev - 1 + vocabList.length) % vocabList.length), 150);

          };


          return (

            <div className="flex flex-col items-center">

              <div className="text-slate-500 mb-4">{index + 1} / {vocabList.length}</div>

              

              <div className="relative w-full max-w-md h-80 perspective-1000 mb-8 cursor-pointer" onClick={() => setIsFlipped(!isFlipped)}>

                <div className={`w-full h-full transition-transform duration-500 transform-style-3d ${isFlipped ? 'rotate-y-180' : ''}`}>

        function Quiz({ data, title }) {

          const [currentQ, setCurrentQ] = useState(0);

          const [score, setScore] = useState(0);

          const [showResult, setShowResult] = useState(false);

          const [selected, setSelected] = useState(null);


          const handleAnswer = (option) => {

            setSelected(option);

            if (option === data[currentQ].answer) setScore(score + 1);

            

            setTimeout(() => {

              setSelected(null);

              if (currentQ < data.length - 1) {

                setCurrentQ(currentQ + 1);

              } else {

                setShowResult(true);

              }

            }, 1000);

          };


          const restart = () => {

            setCurrentQ(0);

            setScore(0);

            setShowResult(false);

            setSelected(null);

          };


          if (showResult) {

            return (

              <div className="bg-white p-8 rounded-2xl shadow-lg text-center max-w-lg mx-auto border-t-4 border-teal-500">

                <h2 className="text-3xl font-bold mb-4">Natija</h2>

                <p className="text-xl mb-6 text-slate-600">Siz {data.length} ta savoldan {score} tasiga to'g'ri javob berdingiz!</p>

                <div className="text-6xl mb-8">

                  {score === data.length ? '🏆' : score > data.length / 2 ? '👍' : '📚'}

                </div>

                <button onClick={restart} className="bg-teal-600 text-white px-8 py-3 rounded-full font-bold flex items-center justify-center gap-2 mx-auto hover:bg-teal-700">

                  <RotateCcw size={20} /> Qayta urinish

                </button>

              </div>

            );

          }


          const q = data[currentQ];


          return (

            <div className="max-w-2xl mx-auto">

              <div className="flex justify-between items-center mb-6">

                <h2 className="text-xl font-bold text-teal-700">{title}</h2>

                <span className="text-sm font-bold bg-teal-100 text-teal-800 px-3 py-1 rounded-full">

                  {currentQ + 1} / {data.length}

                </span>

              </div>


              <div className="bg-white p-8 rounded-2xl shadow-md mb-6 border border-slate-100 text-center">

                <h3 className="text-5xl font-bold text-slate-800 py-8">{q.q}</h3>

              </div>


              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">

                {q.options.map((opt, idx) => {

                  let btnClass = "bg-white border-2 border-slate-200 text-slate-700 hover:border-teal-500 hover:bg-teal-50";

                  if (selected) {

                    if (opt === q.answer) btnClass = "bg-green-100 border-green-500 text-green-800";

                    else if (opt === selected) btnClass = "bg-red-100 border-red-500 text-red-800";

                    else btnClass = "bg-white border-slate-200 text-slate-400 opacity-50";

                  }

                  

                  return (

                    <button

                      key={idx}

                      disabled={selected !== null}

                      onClick={() => handleAnswer(opt)}

                      className={`p-4 rounded-xl font-medium text-lg transition-all flex items-center justify-between ${btnClass}`}

                    >

                      <span>{opt}</span>

                      {selected && opt === q.answer && <CheckCircle2 className="text-green-600" size={20}/>}

                      {selected === opt && opt !== q.answer && <XCircle className="text-red-600" size={20}/>}

                    </button>

                  )

                })}

              </div>

            </div>

          );

        }

        function ExerciseSentence({ index, item, options }) {

          const [val, setVal] = useState("");

          const isCorrect = val === item.answer;


          return (

            <div className="flex items-center flex-wrap gap-2 text-lg">

              <span className="text-slate-400 font-bold mr-2">{index + 1}.</span>

              <span>{item.text[0]}</span>

              <select 

                value={val} 

                onChange={(e) => setVal(e.target.value)}

                className={`border-b-2 outline-none font-bold text-center px-2 py-1 bg-transparent transition-colors cursor-pointer

                  ${val === "" ? "border-slate-300 w-24" : 

                    isCorrect ? "border-green-500 text-green-600" : "border-red-500 text-red-600"}`}

              >

                <option value="" disabled></option>

                {options.map(opt => <option key={opt} value={opt}>{opt}</option>)}

              </select>

              <span>{item.text[1]}</span>

              {val !== "" && (

                isCorrect ? <CheckCircle2 className="text-green-500 ml-2" size={20}/> : <XCircle className="text-red-500 ml-2" size={20}/>

              )}

          );

        }
